  <script>
    // Global state
    let allData = [];
    let selectedPickerId = null;
    let pendingDelete = null;
    const STORAGE_KEY = 'warehouse_data';
    const DATA_VERSION = '1.0';

    const defaultConfig = {
      app_title: '📦 Warehouse Workflow Manager',
      company_name: 'Streamline your warehouse operations',
      primary_color: '#1e3a5f',
      secondary_color: '#ffffff',
      text_color: '#0f172a',
      accent_color: '#2563eb',
      surface_color: '#f1f5f9'
    };

    // Helper functions
    function saveData() {
      localStorage.setItem(STORAGE_KEY, JSON.stringify({
        version: DATA_VERSION,
        timestamp: new Date().toISOString(),
        data: allData
      }));
    }

    function loadData() {
      try {
        const stored = localStorage.getItem(STORAGE_KEY);
        if (stored) {
          const parsed = JSON.parse(stored);
          if (parsed.version === DATA_VERSION) {
            allData = parsed.data || [];
            return;
          }
        }
      } catch (e) {
        console.error('Failed to load data:', e);
      }
      allData = [];
    }

    function addBackendId() {
      return 'id_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
    }

    function getOrders() {
      return allData.filter(d => d.type === 'order');
    }

    function getEmployees() {
      return allData.filter(d => d.type === 'employee');
    }

    function showToast(message) {
      const toast = document.getElementById('toast');
      const toastMessage = document.getElementById('toast-message');
      toastMessage.textContent = message;
      toast.classList.remove('translate-y-20', 'opacity-0');
      setTimeout(() => {
        toast.classList.add('translate-y-20', 'opacity-0');
      }, 3000);
    }

    function showDeleteModal(message, onConfirm) {
      document.getElementById('delete-message').textContent = message;
      document.getElementById('delete-modal').classList.remove('hidden');
      pendingDelete = onConfirm;
      document.getElementById('confirm-delete-btn').onclick = async () => {
        if (pendingDelete) await pendingDelete();
        hideDeleteModal();
      };
    }

    function hideDeleteModal() {
      document.getElementById('delete-modal').classList.add('hidden');
      pendingDelete = null;
    }

    // Tab switching
    function switchTab(tab) {
      ['orders', 'employees', 'analytics', 'employee-metrics'].forEach(t => {
        document.getElementById(`content-${t}`).classList.add('hidden');
        document.getElementById(`tab-${t}`).classList.remove('tab-active');
        document.getElementById(`tab-${t}`).classList.add('text-slate-600', 'hover:bg-slate-100');
      });
      document.getElementById(`content-${tab}`).classList.remove('hidden');
      document.getElementById(`tab-${tab}`).classList.add('tab-active');
      document.getElementById(`tab-${tab}`).classList.remove('text-slate-600', 'hover:bg-slate-100');
      
      if (tab === 'analytics') updateAnalytics();
      if (tab === 'employee-metrics') updateMetricsDashboard();
    }

    // Employee assignment logic
    function getLevelForItemCount(count) {
      if (count <= 10) return 'junior';
      if (count <= 20) return 'intermediate';
      if (count <= 35) return 'senior';
      return 'expert';
    }

    function getLevelPriority(level) {
      const priorities = { junior: 1, intermediate: 2, senior: 3, expert: 4 };
      return priorities[level] || 0;
    }

    function findAvailablePicker(itemCount) {
      const requiredLevel = getLevelForItemCount(itemCount);
      const requiredPriority = getLevelPriority(requiredLevel);
      
      const availablePickers = getEmployees().filter(e => 
        e.employee_role === 'picker' && 
        e.employee_available && 
        e.employee_present
      );

      // Sort by level priority (prefer exact match, then higher levels)
      availablePickers.sort((a, b) => {
        const aPriority = getLevelPriority(a.employee_level);
        const bPriority = getLevelPriority(b.employee_level);
        const aDiff = Math.abs(aPriority - requiredPriority);
        const bDiff = Math.abs(bPriority - requiredPriority);
        if (aDiff !== bDiff) return aDiff - bDiff;
        return bPriority - aPriority; // Prefer higher level if same difference
      });

      return availablePickers.find(p => getLevelPriority(p.employee_level) >= requiredPriority) || availablePickers[0];
    }

    function findAvailableChecker(excludeId) {
      return getEmployees().find(e => 
        e.employee_role === 'checker' && 
        e.employee_available && 
        e.employee_present &&
        e.__backendId !== excludeId
      );
    }

    // Order management
    async function createOrder(invoiceNumber, partyName, itemCount, picker) {
      if (getOrders().length >= 500) {
        showToast('Order limit reached (500). Please complete existing orders.');
        return;
      }

      const order = {
        type: 'order',
        __backendId: addBackendId(),
        invoice_number: invoiceNumber,
        party_name: partyName,
        item_count: itemCount,
        status: 'picking',
        picker_id: picker.__backendId,
        picker_name: picker.employee_name,
        checker_id: '',
        checker_name: '',
        packer_id: '',
        packer_name: '',
        delivery_status: '',
        created_at: new Date().toISOString(),
        completed_at: ''
      };

      allData.push(order);
      saveData();
      renderOrders();
      showToast(`Order ${invoiceNumber} assigned to ${picker.employee_name} for picking`);
    }

    async function updateOrderStatus(order, newStatus) {
      const orderIndex = allData.findIndex(d => d.__backendId === order.__backendId);
      if (orderIndex === -1) return;

      const updates = { ...order, status: newStatus };

      if (newStatus === 'checking') {
        const checker = findAvailableChecker(order.picker_id);
        if (!checker) {
          showToast('No available checker found!');
          return;
        }
        updates.checker_id = checker.__backendId;
        updates.checker_name = checker.employee_name;
      } else if (newStatus === 'packing') {
        updates.packer_id = order.checker_id;
        updates.packer_name = order.checker_name;
      } else if (newStatus === 'delivery') {
        updates.delivery_status = 'pending';
      } else if (newStatus === 'delivered') {
        updates.delivery_status = 'delivered';
        updates.completed_at = new Date().toISOString();
      }

      allData[orderIndex] = updates;
      saveData();
      renderOrders();
      updateAnalytics();
      showToast(`Order ${order.invoice_number} updated to ${newStatus}`);
    }

    // Employee management
    async function addEmployee(name, role, level) {
      if (getEmployees().length >= 499) {
        showToast('Employee limit reached (499).');
        return;
      }

      const employee = {
        type: 'employee',
        __backendId: addBackendId(),
        employee_name: name,
        employee_role: role,
        employee_level: level,
        employee_available: true,
        employee_present: true
      };

      allData.push(employee);
      saveData();
      renderEmployees();
      showToast(`Employee ${name} added successfully`);
    }

    async function toggleEmployeeStatus(employee, field) {
      const index = allData.findIndex(d => d.__backendId === employee.__backendId);
      if (index !== -1) {
        allData[index][field] = !employee[field];
        saveData();
        renderEmployees();
        updateAnalytics();
      }
    }

    async function deleteEmployee(employee) {
      const index = allData.findIndex(d => d.__backendId === employee.__backendId);
      if (index !== -1) {
        allData.splice(index, 1);
        saveData();
        renderEmployees();
        updateAnalytics();
        showToast(`Employee ${employee.employee_name} removed`);
      }
    }

    // Manual assignment
    function showManualAssign() {
      const panel = document.getElementById('manual-assign-panel');
      panel.classList.toggle('hidden');
      if (!panel.classList.contains('hidden')) {
        renderPickerSelection();
      }
    }

    function renderPickerSelection() {
      const container = document.getElementById('picker-selection');
      const pickers = getEmployees().filter(e => e.employee_role === 'picker' && e.employee_present);
      
      if (pickers.length === 0) {
        container.innerHTML = '<p class="col-span-4 text-slate-500">No pickers available. Add employees first.</p>';
        return;
      }

      container.innerHTML = pickers.map(p => `
        <button type="button" onclick="selectPicker('${p.__backendId}')" 
          id="picker-btn-${p.__backendId}"
          class="p-3 border-2 rounded-lg text-left transition-all ${p.employee_available ? 'border-slate-200 hover:border-blue-400' : 'border-red-200 bg-red-50 opacity-60'}">
          <p class="font-medium text-sm">${p.employee_name}</p>
          <p class="text-xs text-slate-500 capitalize">${p.employee_level} • ${p.employee_available ? '✅ Available' : '🔴 Busy'}</p>
        </button>
      `).join('');
    }

    function selectPicker(id) {
      document.querySelectorAll('[id^="picker-btn-"]').forEach(btn => {
        btn.classList.remove('border-blue-500', 'bg-blue-50');
      });
      document.getElementById(`picker-btn-${id}`).classList.add('border-blue-500', 'bg-blue-50');
      selectedPickerId = id;
    }

    async function submitManualOrder() {
      if (!selectedPickerId) {
        showToast('Please select a picker');
        return;
      }

      const invoiceNumber = document.getElementById('invoice-number').value.trim();
      const partyName = document.getElementById('party-name').value.trim();
      const itemCount = parseInt(document.getElementById('item-count').value);

      if (!invoiceNumber || !partyName || !itemCount) {
        showToast('Please fill all fields');
        return;
      }

      const container = document.getElementById('orders-list');

    function getNextAction(order) {
      const btnClass = 'px-3 py-1.5 rounded-lg text-sm font-medium transition-all';
      
      switch(order.status) {
        case 'picking':
          return `<button onclick="advanceOrder('${order.__backendId}', 'checking')" class="${btnClass} bg-blue-600 text-white hover:bg-blue-700">✓ Picking Done</button>`;
        case 'checking':
          return `<button onclick="advanceOrder('${order.__backendId}', 'packing')" class="${btnClass} bg-indigo-600 text-white hover:bg-indigo-700">✓ Checking Done</button>`;
        case 'packing':
          return `<button onclick="advanceOrder('${order.__backendId}', 'delivery')" class="${btnClass} bg-purple-600 text-white hover:bg-purple-700">✓ Packing Done</button>`;
        case 'delivery':
          return `
            <button onclick="advanceOrder('${order.__backendId}', 'delivered')" class="${btnClass} bg-green-600 text-white hover:bg-green-700">✓ Delivered</button>
            <span class="px-2 py-1.5 bg-amber-100 text-amber-700 rounded-lg text-sm">🚚 Out for Delivery</span>
          `;
        default:
          return '';
      }
    }

    // Employee Metrics Dashboard
    function getEmployeeMetrics(employee) {
      const orders = getOrders();
      let taskCount = 0;
      let completedCount = 0;
      let totalItems = 0;

      orders.forEach(order => {
        if (order.picker_id === employee.__backendId) {
          taskCount++;
          if (order.status !== 'picking') completedCount++;
          totalItems += order.item_count;
        } else if (order.checker_id === employee.__backendId) {
          taskCount++;
          if (order.status !== 'checking') completedCount++;
        } else if (order.packer_id === employee.__backendId) {
          taskCount++;
          if (order.status !== 'packing') completedCount++;
        } else if (order.delivery_status === 'delivered' && order.picker_id === employee.__backendId) {
          // Counted in picker metrics
        }
      });

      const successRate = taskCount > 0 ? Math.round((completedCount / taskCount) * 100) : 0;
      const avgItems = taskCount > 0 ? (totalItems / taskCount).toFixed(1) : 0;

      return { taskCount, completedCount, successRate, avgItems };
    }

        return metricsB.completedCount - metricsA.completedCount;
      });

    function renderTrendChart(employees) {
      const container = document.getElementById('trend-chart');
      const orders = getOrders();
      const last7Days = [];
      
      for (let i = 6; i >= 0; i--) {
        const date = new Date();
        date.setDate(date.getDate() - i);
        const dateStr = date.toISOString().split('T')[0];
        last7Days.push(dateStr);
      }

      const dailyStats = {};
      last7Days.forEach(date => {
        dailyStats[date] = { total: 0, completed: 0 };
      });

      orders.forEach(order => {
        const dateStr = order.created_at.split('T')[0];
        if (dailyStats[dateStr]) {
          dailyStats[dateStr].total++;
          if (order.status === 'delivered') {
            dailyStats[dateStr].completed++;
          }
        }
      });

      const maxDaily = Math.max(...Object.values(dailyStats).map(d => d.total), 1);

      container.innerHTML = last7Days.map(date => {
        const day = new Date(date).toLocaleDateString('en-US', { weekday: 'short' });
        const stats = dailyStats[date];
        const completionRate = stats.total > 0 ? Math.round((stats.completed / stats.total) * 100) : 0;
        
        return `
          <div>
            <div class="flex items-center justify-between mb-1">
              <span class="text-sm text-slate-600">${day}</span>
              <span class="text-xs text-slate-500">${stats.completed}/${stats.total}</span>
            </div>
            <div class="flex gap-1">
              <div class="flex-1 h-8 bg-slate-100 rounded-lg overflow-hidden">
                <div class="h-full bg-gradient-to-r from-green-400 to-green-600" style="width: ${stats.total > 0 ? (stats.completed / stats.total) * 100 : 0}%"></div>
              </div>
              <div class="text-right w-12 text-xs font-bold text-slate-700 flex items-center justify-end">${completionRate}%</div>
            </div>
          </div>
        `;
      }).join('');
    }

      const blob = new Blob([csv], { type: 'text/csv' });

      a.href = url;
      a.download = `employee-metrics-${new Date().toISOString().split('T')[0]}.csv`;
      a.click();
      showToast('Metrics exported successfully!');
    }
